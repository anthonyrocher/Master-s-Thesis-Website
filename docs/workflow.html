JMA records include all seismic activity observations. They are to be converted to csv first, ensuring that they are well encoded.

Records include of course latitude and longitude in DMS format. Then the following code convert those variables in the CRS WGS84 (epsg:4326) format:

# Convert latitude_deg / lat and longitude_deg / lon in DMS to WGS84 (epsg:4326)
def dms_to_decimals(latlon_deg: int = 31, latlon: int = 1176):
    minute = latlon // 100
    second = latlon % 100
    return latlon_deg + minute / 60 + second / 3600
  
jma_damage["latitude"] = jma_damage.apply(
    lambda row : dms_to_decimals(row["latitude_deg"], row["lat"]),
    axis = 1)

jma_damage["longitude"] = jma_damage.apply(
    lambda row : dms_to_decimals(row["longitude_deg"], row["lon"]),
    axis = 1)

The records include a “damage _class” variable where damages are recorded.

********************************************************************************
* FILLING IN MISSING VALUES

* Linear interpolation
* Those for which 2015 and 2020 are available
foreach var of varlist pop_total median_age lifeexp_male lifeexp_female pop_u15 pop_o65 did_pop foreigners households pop_total2 employed unemployed labor_force skilled_workers selfemp_with selfemp_without commute_in commute_out {

    * Create base
    gen `var'_base = .
    bysort areacode (year): replace `var'_base = `var'[1]

    * Linear interpolation coefficients
    gen `var'_delta = .
    bysort areacode (year): replace `var'_delta = (`var'[6] / `var'[1]) ^ (1/5)

    * Interpolation computation
    gen `var'_linear = .
    forvalues y = 2015/2023 {
        local t = `y' - 2015
        replace `var'_linear = `var'_base * (`var'_delta ^ `t') if year == `y'
    }

    * Drop temporary var
    drop `var'_delta
    drop `var'_base
}

* Regression
gen inmigrants_reg = inmigrants
gen outmigrants_reg = outmigrants
foreach var in inmigrants outmigrants {

    gen yhat = .

    levelsof areacode, local(areas)

    foreach a of local areas {
        count if areacode == `a' & !missing(year, `var')
        
        if r(N) >= 2 {
            regress `var' year if areacode == `a'
            predict temp if areacode == `a'
            
            replace yhat = round(temp) if areacode == `a'
            drop temp
        }
        else {
            display "Skipping areacode `a' for `var' (not enough data)"
        }
    }

    replace `var'_reg = yhat if missing(`var')
    drop yhat
}


* Take the nearest value
foreach var of varlist comm_centers libraries hospitals physicians dentists  {

    bysort areacode (year): replace `var'_near = `var'_near[_n-1] if missing(`var'_near) 
}

foreach var of varlist fin_power {
